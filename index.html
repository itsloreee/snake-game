Use arrow keys or buttons to move!

body { background: #000; color: white; text-align: center; font-family: Arial; } canvas { background: #111; border: 3px solid white; margin-top: 10px; } .controls button { font-size: 20px; margin: 5px; padding: 10px; } let canvas = document.getElementById("game"); let ctx = canvas.getContext("2d"); let snake = [{ x: 200, y: 200 }]; let food = { x: 100, y: 100 }; let dx = 20; let dy = 0; let score = 0; let speed = 120; let gameRunning = true; document.addEventListener("keydown", changeDirection); function changeDirection(e) { if (e.key === "ArrowUp" && dy === 0) { dx = 0; dy = -20; } if (e.key === "ArrowDown" && dy === 0) { dx = 0; dy = 20; } if (e.key === "ArrowLeft" && dx === 0) { dx = -20; dy = 0; } if (e.key === "ArrowRight" && dx === 0) { dx = 20; dy = 0; } } function setDir(dir) { if (dir === "up" && dy === 0) { dx = 0; dy = -20; } if (dir === "down" && dy === 0) { dx = 0; dy = 20; } if (dir === "left" && dx === 0) { dx = -20; dy = 0; } if (dir === "right" && dx === 0) { dx = 20; dy = 0; } } function gameLoop() { if (!gameRunning) return; ctx.clearRect(0, 0, 400, 400); let head = { x: snake[0].x + dx, y: snake[0].y + dy }; if (head.x < 0 || head.x >= 400 || head.y < 0 || head.y >= 400) { gameOver(); return; } for (let part of snake) { if (head.x === part.x && head.y === part.y) { gameOver(); return; } } snake.unshift(head); if (head.x === food.x && head.y === food.y) { score++; document.getElementById("score").innerText = score; food = { x: Math.floor(Math.random() * 20) * 20, y: Math.floor(Math.random() * 20) * 20 }; if (speed > 50) { speed -= 5; clearInterval(loop); loop = setInterval(gameLoop, speed); } } else { snake.pop(); } ctx.fillStyle = "lime"; snake.forEach((part) => { ctx.fillRect(part.x, part.y, 20, 20); }); ctx.fillStyle = "red"; ctx.fillRect(food.x, food.y, 20, 20); } function gameOver() { gameRunning = false; document.getElementById("gameover").innerText = "Game Over!"; } let loop = setInterval(gameLoop, speed);
